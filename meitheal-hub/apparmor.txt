#include <tunables/global>

profile meitheal_hub flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # ── Capabilities ──
  capability chown,
  capability dac_override,
  capability setuid,
  capability setgid,
  capability net_bind_service,

  # ── Network ──
  network inet tcp,
  network inet udp,
  network inet6 tcp,
  network inet6 udp,
  deny network raw,
  deny network packet,

  # ── Node.js runtime ──
  /usr/bin/node ix,
  /usr/local/bin/node ix,

  # ── Application files (read-only) ──
  /opt/meitheal/** r,
  /opt/meitheal/node_modules/.pnpm/**/node_modules/** r,

  # ── Entrypoints ──
  /run.sh ix,
  /run-local.sh ix,

  # ── Persistent data (read/write) ──
  /data/ r,
  /data/** rwk,

  # ── Temp files ──
  /tmp/ r,
  /tmp/** rwk,

  # ── Deny shell access ──
  deny /bin/dash mrwklx,
  deny /bin/sh mrwklx,
  deny /usr/bin/top mrwklx,

  # ── Deny mount ──
  deny mount,

  # ── Deny system path writes ──
  deny /boot/** wl,
  deny /root/** wl,
  deny /srv/** wl,

  # ── Proc/sys restrictions ──
  deny @{PROC}/sysrq-trigger rwklx,
  deny @{PROC}/mem rwklx,
  deny @{PROC}/kmem rwklx,
  deny @{PROC}/kcore rwklx,
  deny /sys/firmware/** rwklx,
  deny /sys/kernel/security/** rwklx,
}
